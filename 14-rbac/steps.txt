## RBAC
minikube start --vm-driver=none  --extra-config=apiserver.authorization-mode=RBAC

## Create keys and sign
openssl genrsa -out oscar.key 2048
openssl req -new -key oscar.key -out oscar.csr -subj "/CN=oscar/O=dev"
sudo openssl x509 -req -in oscar.csr -CA /home/oscarock/.minikube/ca.crt -CAkey /home/oscarock/.minikube/ca.key -CAcreateserial -out oscar.crt -days 500
openssl x509 -in  oscar.crt  -noout -text

## Isolated env
kubectl config view  | grep server
docker run --rm -ti --network=minikube -v $PWD:/test -w /test  -v /home/oscarock/.minikube/ca.crt:/ca.crt -v /usr/local/bin/kubectl:/usr/bin/kubectl alpine sh

## Configure kubectl for user
kubectl config set-cluster minikube --server=https://192.168.49.2:8443 --certificate-authority=/ca.crt
kubectl config set-credentials oscar --client-certificate=oscar.crt --client-key=oscar.key
kubectl config set-context oscar --cluster=minikube --user=oscar
kubectl config use-context oscar
